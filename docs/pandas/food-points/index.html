def get_food_points(food):
    if('chicken' in food):
        return 2
    
    if('stake' in food):
        return 1
    
    if('burger' in food):
        return 1

    if('salad' in food):
        return 7
    
    return None

df['food_points'] = df['dinner'].apply(get_food_points)
