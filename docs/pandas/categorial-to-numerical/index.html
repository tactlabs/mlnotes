def get_medal_points(medal):
    if(medal == 'gold'):
        return 10
    
    if(medal == 'silver'):
        return 5
    
    if(medal == 'bronze'):
        return 1
    
    return 0

df['medal_points'] = df['medal'].apply(get_medal_points)
