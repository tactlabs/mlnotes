Find-two-digits-with-spaces

import re
text = "It happened on Feb 21 at 3:30"

answer= re.findall(r'\b\d{2}\b', text)
print(answer)
['21', '30']

# To match only two digits with spaces
answers = re.findall(r'\s\d{2}\s', text)
for a in answers:
    print(a.strip())
21
